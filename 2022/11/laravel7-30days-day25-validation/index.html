    public function store(Request $request)
    {
        // 加這個 !!!
        Validator::make($request->all(), [
            'title' => 'required'
        ])->validate();
        
        Auth::user()->tasks()->create([
            'title' => $request->title
        ]);
        
        return redirect()->route('tasks.index');   
    }
